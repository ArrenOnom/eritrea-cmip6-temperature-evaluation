// =========================================================================
// GEE SCRIPT: HISTORICAL GEOTIFF EXPORTS (31 VALID CMIP6 MODELS + ERA5)
// Dedicated Export Folder: CMIP6_Eritrea_Historical_Rasters
// =========================================================================

var aez = ee.FeatureCollection("projects/disaster-management-474719/assets/SELECTED_ZONES_WGS1984");
var startYear = 1981;
var endYear = 2014;
var spatialScale = 27830; // Native 0.25° grid size

// -------------------------------------------------------------------------
// CUSTOM EXPORT FOLDER NAME
// -------------------------------------------------------------------------
var exportFolder = 'CMIP6_Eritrea_Historical_Rasters';

Map.centerObject(aez, 7);
Map.addLayer(aez, {color: 'red'}, 'AEZ Boundaries');

// -------------------------------------------------------------------------
// 1. ERA5 REANALYSIS BASELINE EXPORTS (3 GeoTIFFs)
// -------------------------------------------------------------------------
var era5Col = ee.ImageCollection("ECMWF/ERA5/MONTHLY")
  .filterDate(startYear + '-01-01', (endYear + 1) + '-01-01')
  .filterBounds(aez);

var era5Vars = [
  {geeVar: 'minimum_2m_air_temperature', name: 'Tmin'},
  {geeVar: 'mean_2m_air_temperature',    name: 'Tmean'},
  {geeVar: 'maximum_2m_air_temperature', name: 'Tmax'}
];

era5Vars.forEach(function(v) {
  var era5Mean = era5Col
    .select(v.geeVar)
    .mean()
    .subtract(273.15)
    .rename(v.name)
    .clip(aez);

  Export.image.toDrive({
    image: era5Mean,
    description: 'ERA5_Reanalysis_Historical_1981_2014_' + v.name,
    folder: exportFolder,
    scale: spatialScale,
    region: aez.geometry().bounds(),
    crs: 'EPSG:4326',
    maxPixels: 1e9
  });
});

// -------------------------------------------------------------------------
// 2. 31 VERIFIED CMIP6 MODELS (93 GeoTIFFs)
// -------------------------------------------------------------------------
var validModels = [
  'ACCESS-CM2', 'ACCESS-ESM1-5', 'BCC-CSM2-MR', 'CanESM5', 
  'CMCC-CM2-SR5', 'CMCC-ESM2', 'CNRM-CM6-1', 'CNRM-ESM2-1', 
  'EC-Earth3', 'EC-Earth3-Veg-LR', 'FGOALS-g3', 'GFDL-CM4', 'GFDL-ESM4', 
  'GISS-E2-1-G', 'HadGEM3-GC31-LL', 'HadGEM3-GC31-MM', 'INM-CM4-8', 
  'INM-CM5-0', 'IPSL-CM6A-LR', 'KACE-1-0-G', 'KIOST-ESM', 'MIROC-ES2L', 
  'MIROC6', 'MPI-ESM1-2-HR', 'MPI-ESM1-2-LR', 'MRI-ESM2-0', 'NESM3', 
  'NorESM2-LM', 'NorESM2-MM', 'TaiESM1', 'UKESM1-0-LL'
];

var cmipVars = [
  {geeVar: 'tasmin', name: 'Tmin'},
  {geeVar: 'tas',    name: 'Tmean'},
  {geeVar: 'tasmax', name: 'Tmax'}
];

validModels.forEach(function(modelName) {
  var modelCol = ee.ImageCollection("NASA/GDDP-CMIP6")
    .filter(ee.Filter.eq('model', modelName))
    .filter(ee.Filter.eq('scenario', 'historical'))
    .filterDate(startYear + '-01-01', (endYear + 1) + '-01-01')
    .filterBounds(aez);

  cmipVars.forEach(function(v) {
    var meanImg = modelCol
      .select(v.geeVar)
      .mean()
      .subtract(273.15)
      .rename(v.name)
      .clip(aez);

    Export.image.toDrive({
      image: meanImg,
      description: 'CMIP6_' + modelName.replace(/-/g, '_') + '_Historical_1981_2014_' + v.name,
      folder: exportFolder,
      scale: spatialScale,
      region: aez.geometry().bounds(),
      crs: 'EPSG:4326',
      maxPixels: 1e9
    });
  });
});
